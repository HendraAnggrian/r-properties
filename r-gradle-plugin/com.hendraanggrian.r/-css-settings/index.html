charset

The charset to be used in case neither a charset rule nor a BOM is present. Default value is StandardCharsets.UTF_8.

var charset: Charset

cssVersion

The CSS version to use. Default value is ECSSVersion.CSS30.

var cssVersion: ECSSVersion

isWriteClassSelector

Determine whether adapter should write class selector (e.g.: .box { ... }) It is enabled by default.

var isWriteClassSelector: Boolean

isWriteElementTypeSelector

Determine whether adapter should write element type selector (e.g.: ul { ... }) It is disabled by default.

var isWriteElementTypeSelector: Boolean

isWriteIdSelector

Determine whether adapter should write ID selector (e.g.: #container { ... }) It is enabled by default.

var isWriteIdSelector: Boolean
